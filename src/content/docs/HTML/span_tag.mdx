```js
<!-- Basic inline span -->
<p>This is <span>inline text</span> inside a paragraph.</p>

<!-- Span with class for styling -->
<p>Welcome, <span class="username">Alice</span>!</p>

<!-- Span with id for JS targeting -->
<span id="highlight">Highlighted text</span>

<!-- Inline styling -->
<p><span style="color: red;">Red text</span> and <span style="font-weight: bold;">bold text</span></p>
```
